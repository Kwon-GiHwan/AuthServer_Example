FROM python:3.10.0-alpine

WORKDIR /app  # /usr/src/app으로 이동
ADD . /app
COPY . .

#RUN apk add install mysql-devel gcc


RUN apk update \
    && apk add --virtual build-deps gcc python3-dev musl-dev \
    && apk add --no-cache mariadb-dev\
    && apk del build-deps

#RUN pip install --no-cache-dir -r requirements.txt  # requirements.txt 내용 실행

RUN curl -sSL https://install.python-poetry.org | python3 -
ENV PATH="/root/.poetry/bin:${PATH}"

RUN pip install poetry


# Project initialization:
RUN POETRY_VIRTUALENVS_CREATE=false \
  && poetry install $(test "$YOUR_ENV" == production && echo "--no-dev") --no-interaction --no-ansi --no-root


EXPOSE 80
